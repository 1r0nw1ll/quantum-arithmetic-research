from fractions import Fraction
from decimal import Decimal, getcontext
import matplotlib.pyplot as plt

getcontext().prec = 80  # high-precision reference

def factorial(n:int)->int:
    f=1
    for k in range(2,n+1): f*=k
    return f

def rational_exp(x: Fraction, n_terms: int) -> Fraction:
    s = Fraction(0,1); x_pow = Fraction(1,1)
    for k in range(n_terms+1):
        s += x_pow / factorial(k)
        x_pow *= x
    return s

def rational_sin(x: Fraction, n_terms: int) -> Fraction:
    s = Fraction(0,1); x_pow = x; sign = 1
    for k in range(n_terms+1):
        s += Fraction(sign,1) * x_pow / factorial(2*k+1)
        x_pow *= x*x; sign *= -1
    return s

def frac_to_decimal(fr: Fraction) -> Decimal:
    return Decimal(fr.numerator) / Decimal(fr.denominator)

def dec_exp(x: Decimal) -> Decimal: return x.exp()

def dec_sin(x: Decimal, terms: int = 100) -> Decimal:
    s = Decimal(0); x_pow = x; sign = Decimal(1)
    for k in range(terms):
        s += sign * (x_pow / Decimal(factorial(2*k+1)))
        x_pow = x_pow * x * x; sign = -sign
    return s

# Canonical QA tuple and sampling point
def qa_tuple(b:int,e:int): return (b,e,b+e,b+2*e)
def qa_x_from_tuple(b:int,e:int) -> Fraction:
    b,e,d,a = qa_tuple(b,e); return Fraction(e,d)

Instead of relying on the inherent, and sometimes ambiguous, structure of real-world datasets, the Lancichinetti–Fortunato–Radicchi (LFR) benchmark allows us to generate synthetic graphs where we can precisely control the properties of the community structure.
The Core Concept: The Mixing Parameter (μ)
The most important parameter for our experiment is the mixing parameter (mu, μ). It defines the fraction of a node's edges that connect to nodes outside its own community.
    • Low μ (e.g., 0.1): Communities are dense and well-defined. 90% of each node's connections are internal to its community. This should be an easy task for your method.
    • High μ (e.g., 0.7): Communities are noisy and heavily overlapping. 70% of each node's connections are to other communities. This is a very difficult task, where the "signal" of the structure is weak.
By systematically increasing μ, we can pinpoint the exact breaking point of the QA-aligned structural analysis paradigm.
The Experiment: Plotting Performance vs. Difficulty
Our experiment will be to run the Louvain analysis on a series of LFR graphs with increasing μ values and plot the Adjusted Rand Index (ARI) at each step.
Our Hypothesis: The QA/Louvain method will achieve a near-perfect ARI for low μ values. As μ increases, the ARI will remain high and then drop sharply once the community structure becomes too ambiguous for the modularity-based algorithm to resolve. This will reveal the robustness of the method to noise.

    import networkx as nx
import matplotlib.pyplot as plt
from sklearn.metrics import adjusted_rand_score

def run_lfr_experiment(mu_value, n=250, avg_degree=5, min_community=20):
    """
    Generates an LFR graph for a given mu, runs Louvain, and returns the ARI score.
    """
    # Generate the LFR benchmark graph
    # tau1=3: degree distribution exponent
    # tau2=1.5: community size distribution exponent
    G = nx.LFR_benchmark_graph(
        n=n,
        tau1=3,
        tau2=1.5,
        mu=mu_value,
        average_degree=avg_degree,
        min_community=min_community,
        seed=42
    )

    # The LFR generator stores the ground truth communities on the nodes
    ground_truth_communities = {frozenset(G.nodes[i]['community']) for i in G}
    # Create a partition map {node: community_id} for ARI calculation
    ground_truth_partition = {}
    for i, comm in enumerate(ground_truth_communities):
        for node in comm:
            ground_truth_partition[node] = i

    # --- Apply the Structural Analysis ---
    detected_partition = community_louvain.best_partition(G, random_state=42)

    # --- Evaluate the Results ---
    ground_truth_labels = []
    detected_labels = []
    for node in sorted(G.nodes()):
        ground_truth_labels.append(ground_truth_partition[node])
        detected_labels.append(detected_partition[node])

    ari = adjusted_rand_score(ground_truth_labels, detected_labels)
    
    num_gt_comm = len(ground_truth_communities)
    num_det_comm = len(set(detected_partition.values()))
    
    print(f"mu={mu_value:.2f} | GT Communities={num_gt_comm} | Detected={num_det_comm} | ARI={ari:.4f}")
    
    return ari

# --- Main Execution ---
if __name__ == "__main__":

def generate_sierpinski_graph(level):
    """
    Recursively generates a Sierpinski gasket graph.
    """
    if level == 0:
        # Base case: A single triangle
        G = nx.Graph()
        G.add_edges_from([(0, 1), (1, 2), (2, 0)])
        return G
    else:
        # Recursive step:
        # 1. Get the graph from the previous level
        G_prev = generate_sierpinski_graph(level - 1)
        
        # 2. Create three copies of the previous level's graph
        G1 = G_prev.copy()
        G2 = G_prev.copy()
        G3 = G_prev.copy()
        
        # 3. Relabel nodes to avoid overlap.
        # The number of nodes in G_prev helps us create non-overlapping labels.
        num_nodes_prev = G_prev.number_of_nodes()
        # G2 nodes will be num_nodes_prev, num_nodes_prev+1, ...
        G2 = nx.relabel_nodes(G2, lambda x: x + num_nodes_prev)
        # G3 nodes will be 2*num_nodes_prev, 2*num_nodes_prev+1, ...
        G3 = nx.relabel_nodes(G3, lambda x: x + 2 * num_nodes_prev)
        
        # 4. Combine the three copies
        G = nx.compose_many([G1, G2, G3])
        
        # 5. Connect the corners of the three copies to form the larger triangle
        # Corner nodes of the original base triangle are 0, 1, 2.
        # We connect corner 1 of G1 to corner 0 of G2.
        G.add_edge(1, 0 + num_nodes_prev)
        # We connect corner 2 of G1 to corner 0 of G3.
        G.add_edge(2, 0 + 2 * num_nodes_prev)
        # We connect corner 2 of G2 to corner 1 of G3.
        G.add_edge(2 + num_nodes_prev, 1 + 2 * num_nodes_prev)
        
        return G

The Multi-Resolution Method
This is a powerful technique in network science. Here's how it works:
    1. First Pass (Micro-Communities): We run the standard Louvain algorithm on the original graph. As we've seen, this finds the small, ultra-dense "micro-communities" (the 18 communities in our case).
    2. Construct a Super-Graph: We create a new, smaller graph where each node represents one of the micro-communities found in the first pass. We draw a weighted edge between two of these new "super-nodes" if there are connections between the original communities. The weight of the edge is the number of connections. This is called a "quotient graph."
    3. Second Pass (Macro-Communities): We run the Louvain algorithm again, but this time on the new, smaller, weighted super-graph. This second pass is blind to the fine details and can only see the high-level structure—the connections between the clusters.
Our hypothesis is that this second pass will correctly identify the 3 macro-structures of the Sierpinski fractal.

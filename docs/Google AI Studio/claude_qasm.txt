                if abs(Q) < 0.1:

                elif abs(Q + 1) < 0.1:

            else:  # Singlet
                if abs(Q + 1) < 0.1:
                    return 'charged lepton singlet'
        
        elif color == 'triplet':  # Quarks
            if abs(T3) > 0.4:  # Doublet member
                if abs(Q - 2/3) < 0.1:
                    return 'up-type quark (u/c/t)'
                elif abs(Q + 1/3) < 0.1:
                    return 'down-type quark (d/s/b)'
            else:  # Singlet
                if abs(Q - 2/3) < 0.1:
                    return 'up-type quark singlet'
                elif abs(Q + 1/3) < 0.1:
                    return 'down-type quark singlet'

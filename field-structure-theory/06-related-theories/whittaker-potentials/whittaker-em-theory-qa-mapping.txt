import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

# Define QA scalar harmonic basis functions (modulated F and G)
def QA_scalar_F(theta, phi):
    return np.sin(3 * theta) * np.cos(2 * phi) + np.cos(5 * theta)

def QA_scalar_G(theta, phi):
    return np.sin(4 * phi) * np.cos(3 * theta) + np.sin(2 * theta)

# Compute toroidal coordinates
def toroidal_coordinates(R, r, theta, phi):
    x = (R + r * np.cos(theta)) * np.cos(phi)
    y = (R + r * np.cos(theta)) * np.sin(phi)
    z = r * np.sin(theta)
    return x, y, z

# Set torus parameters and discretization
R = 3  # Major radius
r = 1  # Minor radius

n_theta = 100
n_phi = 100
theta = np.linspace(0, 2 * np.pi, n_theta)
phi = np.linspace(0, 2 * np.pi, n_phi)

Theta, Phi = np.meshgrid(theta, phi)

# Compute field amplitudes from QA-modulated scalar functions
F = QA_scalar_F(Theta, Phi)
G = QA_scalar_G(Theta, Phi)

# Combine for EM-like interpretation (e.g., electric = grad(F), magnetic = grad(G))
E_field_strength = np.abs(F)
B_field_strength = np.abs(G)

# Compute toroidal coordinates
X, Y, Z = toroidal_coordinates(R, r, Theta, Phi)

# Visualize fields on toroidal surface
fig = plt.figure(figsize=(12, 6))
ax1 = fig.add_subplot(121, projection='3d')
ax1.plot_surface(X, Y, Z, facecolors=plt.cm.viridis(E_field_strength / E_field_strength.max()), rstride=1, cstride=1, antialiased=False)
ax1.set_title('QA-Scalar Electric Field (|F|)')

ax2 = fig.add_subplot(122, projection='3d')
ax2.plot_surface(X, Y, Z, facecolors=plt.cm.plasma(B_field_strength / B_field_strength.max()), rstride=1, cstride=1, antialiased=False)
ax2.set_title('QA-Scalar Magnetic Field (|G|)')

plt.tight_layout()
plt.show()

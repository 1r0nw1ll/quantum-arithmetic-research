>>  return math.comb(2*m, m) * math.comb(2*n, n) // math.comb(m + n, n)

>> @lru_cache(maxsize=None)
>> def omega(m: int, n: int) -> float:
>>  """Circular (normalized) Super-Catalan Ω(m,n)."""
>>  return S(m, n) / (4 ** ((m + n) / 2))
>> 
>> # ---------------------------------------------------------------------
>> # 2. Modular arithmetic versions for finite fields ℤ_p
>> # ---------------------------------------------------------------------
>> 
>> def modinv(a: int, p: int) -> int:
>>  """Modular inverse (a⁻¹ mod p)."""
>>  return pow(a, -1, p)
>> 
>> def factorial_mod(k: int, p: int) -> int:
>>  """Factorial modulo p (safe for moderate p)."""
>>  r = 1
>>  for i in range(2, k + 1):
>>  r = (r * i) % p
>>  return r
>> 
>> def comb_mod(n: int, k: int, p: int) -> int:
>>  """n choose k mod p using factorials."""
>>  if k < 0 or k > n:
>>  return 0
>>  num = factorial_mod(n, p)
>>  den = (factorial_mod(k, p) * factorial_mod(n - k, p)) % p
>>  return (num * modinv(den, p)) % p
>> 
>> def S_modp(m: int, n: int, p: int) -> int:
>>  """Super-Catalan S(m,n) mod p."""
>>  s = (comb_mod(2*m, m, p) * comb_mod(2*n, n, p)) % p
>>  s = (s * modinv(comb_mod(m + n, n, p), p)) % p
>>  return s
>> 
>> def omega_modp(m: int, n: int, p: int) -> int:
>>  """
>>  Ωₚ(m,n) = S(m,n) / 4^((m+n)/2) mod p
>>  Integer version of circular Super-Catalan over finite field.
>>  """
>>  s = S_modp(m, n, p)
>>  exp = (m + n) // 2
>>  denom = pow(4, exp, p)
>>  return (s * modinv(denom, p)) % p
>> 
>> # ---------------------------------------------------------------------
>> # 3. QA-pipeline integration helpers
>> # ---------------------------------------------------------------------
>> 
>> def supercatalan_weight(b: int, e: int, mod: int = 24) -> float:
>>  """
>>  Returns normalized QA harmonic weight for tuple (b,e) scaled
>>  by circular Super-Catalan Ω(b,e) within the mod-24/72/144 cycle.
>>  """
>>  ω = omega(b, e)
>>  if mod in (24, 72, 144):
>>  return (ω % 1.0) # harmonic fraction for visualization
>>  return ω
>> 
>> def supercatalan_weight_modp(b: int, e: int, p: int) -> int:
>>  """Finite-field weight (Ω mod p) used in QA modular CSV emitters."""
>>  return omega_modp(b, e, p)
>> 
>> # ---------------------------------------------------------------------
>> # 4. Quick self-test
>> # ---------------------------------------------------------------------
>> if __name__ == "__main__":
>>  for m, n in [(1,1), (2,1), (2,2), (3,2)]:
>>  print(f"S({m},{n}) = {S(m,n):>6}, Ω({m},{n}) = {omega(m,n):.6f}")
>>  print("Ωₚ(3,2,p=13) =", omega_modp(3,2,13))
>> ```

>> ## 🧩 File 2 — `qa_fourier_modq.py`
>> 
>> ```python
>> #!/usr/bin/env python3
>> # qa_fourier_modq.py
>> # ---------------------------------------------------------------------
>> # Quantum Arithmetic × Universal Hyperbolic Geometry
>> # Finite-Field Fourier Summation over ℤ_p (Wildberger–Limanta 2021)
>> #
>> # Provides three chromogeometric Fourier kernels ψ_{c,p}
>> # • ψ_blue (Euclidean / ellipse)
>> # • ψ_red (hyperbolic / hyperbola)
>> # • ψ_green (null / parabola)
>> # Each returns modular harmonic coefficients using Super-Catalan weights.
>> # ---------------------------------------------------------------------
>> 
>> from qa_supercatalan import omega_modp
>> from typing import Literal
>> 
>> Geometry = Literal["blue", "red", "green"]
>> 
>> # ---------------------------------------------------------------------
>> # 1. Fourier kernels
>> # ---------------------------------------------------------------------
>> 
>> def psi_blue(k: int, l: int, p: int) -> int:
>>  """ψ_blue(α^{2k}β^{2l}) = Ωₚ(k,l) (Euclidean)"""
>>  return omega_modp(k // 2, l // 2, p) if k % 2 == 0 and l % 2 == 0 else 0
>> 
>> def psi_red(k: int, l: int, p: int) -> int:
>>  """ψ_red(α^{2k}β^{2l}) = (−1)^{l/2} Ωₚ(k,l) (Hyperbolic)"""
>>  if k % 2 or l % 2:
>>  return 0
>>  val = omega_modp(k // 2, l // 2, p)
>>  return ((-1) ** (l // 2)) * val % p
>> 
>> def psi_green(k: int, l: int, p: int) -> int:
>>  """ψ_green(α^{2k}β^{2l}) = δ_{k,l} (Null / Parabolic)"""
>>  return 1 if k == l else 0
>> 
>> # ---------------------------------------------------------------------
>> # 2. Unified wrapper
>> # ---------------------------------------------------------------------
>> 
>> def psi_cq(k: int, l: int, p: int, geom: Geometry = "blue") -> int:
>>  """General chromogeometric Fourier kernel ψ_{geom,p}."""
>>  if geom == "blue":
>>  return psi_blue(k, l, p)
>>  if geom == "red":
>>  return psi_red(k, l, p)
>>  if geom == "green":
>>  return psi_green(k, l, p)
>>  raise ValueError("geom must be 'blue', 'red', or 'green'")
>> 
>> # ---------------------------------------------------------------------
>> # 3. QA integration: harmonic CSV hooks
>> # ---------------------------------------------------------------------
>> 
>> def fourier_weight_tuple(b: int, e: int, p: int, geom: Geometry = "blue") -> int:
>>  """
>>  Compute per-tuple Fourier weight for (b,e) in QA mod-p engine.
>>  Used by qa_mod_res_engine_hyperbolic.py emitters.
>>  """
>>  return psi_cq(2*b, 2*e, p, geom)
>> 
>> # ---------------------------------------------------------------------
>> # 4. Self-test
>> # ---------------------------------------------------------------------
>> if __name__ == "__main__":
>>  for geom in ("blue", "red", "green"):
>>  vals = [psi_cq(2*i, 2*j, 13, geom) for i in range(4) for j in range(4)]
>>  print(f"{geom:>6}:", vals[:8])
>> ```

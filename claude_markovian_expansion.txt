        print(f"p-value: {p_value:.6f}")
        print(f"Cohen's d: {effect_size:.4f}")
        
        if p_value < 0.05:
            print("✓ SIGNIFICANT: Divisors show enhanced alignment!")
        else:
            print("✗ NOT SIGNIFICANT: No divisor enhancement detected")
            
        return divisors.mean(), non_divisors.mean(), p_value
    
    def convergence_scaling(self):
        """Analyze how convergence time scales with modulus"""
        x = self.df['modulus'].values
        y = self.df['transitionTime'].values
        
        # Test multiple models
        def linear(m, a, b): return a * m + b
        def sqrt(m, a, b): return a * np.sqrt(m) + b
        def log(m, a, b): return a * np.log(m) + b
        
        models = {
            'Linear': linear,
            'Square Root': sqrt,
            'Logarithmic': log
        }
        
        best_model = None
        best_r2 = -np.inf
        
        print("\n=== Convergence Time Scaling ===")
        for name, func in models.items():
            params, _ = curve_fit(func, x, y)
            y_pred = func(x, *params)
            r2 = 1 - np.sum((y - y_pred)**2) / np.sum((y - y.mean())**2)
            print(f"{name}: R² = {r2:.4f}, params = {params}")
            
            if r2 > best_r2:
                best_r2 = r2
                best_model = name
        
        print(f"\n✓ Best fit: {best_model} (R² = {best_r2:.4f})")
        return best_model, best_r2

    """

        super().__init__()
        self.embed_dim = embed_dim
        self.num_heads = num_heads
        self.head_dim = embed_dim // num_heads
        self.geometry = geometry

        self.q_proj = nn.Linear(embed_dim, embed_dim)
        self.k_proj = nn.Linear(embed_dim, embed_dim)
        self.v_proj = nn.Linear(embed_dim, embed_dim)

        # Linear projections and reshape for multi-head
        Q = self.q_proj(query).view(batch_size, -1, self.num_heads, self.head_dim)
        K = self.k_proj(key).view(batch_size, -1, self.num_heads, self.head_dim)
        V = self.v_proj(value).view(batch_size, -1, self.num_heads, self.head_dim)

        spread_scores = self.compute_spread_matrix(Q, K)

        # Softmax over spreads (not raw dot products)
        attn_weights = F.softmax(spread_scores, dim=-1)
        
        # Apply attention to values
        output = torch.matmul(attn_weights, V)
        
        # Concatenate heads and project
        output = output.transpose(1, 2).contiguous().view(
            batch_size, -1, self.embed_dim
        )
        
        return output, attn_weights
    
    def compute_spread_matrix(self, Q, K):
        """
        Compute pairwise spread s_B(q_i, k_j) for all query-key pairs
        
        Spread: s(Q, K) = 1 - [(Q·_B K)² / (Q_B(Q) * Q_B(K))]
        """
        # Compute B-dot products
        QK = torch.einsum('bqhd,bkhd->bqkh', 
                          Q @ self.B, K)  # (batch, queries, keys, heads)
        
        # Compute B-quadrances
        Q_quad = torch.einsum('bqhd,bqhd->bqh', 
                              Q @ self.B, Q).unsqueeze(2)  # (batch, queries, 1, heads)
        K_quad = torch.einsum('bkhd,bkhd->bkh', 
                              K @ self.B, K).unsqueeze(1)  # (batch, 1, keys, heads)
        
        # Spread formula
        spreads = 1 - (QK ** 2) / (Q_quad * K_quad + 1e-8)
        
        return spreads

                edge_weight = spreads[edge_color]

                    graph.add_edge(i, j, 
                                  color=edge_color,
                                  weight=edge_weight)

        Check if dependency graph satisfies QA theorems

        euler_valid = self.check_euler_alignment(graph)
        
        # Pythagorean closure: edge weights satisfy triangle laws
        pythagoras_valid = self.check_pythagoras_edges(graph)
        
        # Chromatic orthogonality: different edge types are separable
        orthogonal = self.check_chromatic_separation(graph)
        
        return euler_valid and pythagoras_valid and orthogonal

class ChromaticReadingComprehension(nn.Module):
    """
    Reading comprehension via QA-geometric reasoning chains
    """
    def __init__(self, encoder):
        super().__init__()
        self.encoder = encoder
        
        # Reasoning layers (one per hop)
        self.reasoning_layers = nn.ModuleList([
            QAReasoningLayer() for _ in range(4)  # 4-hop reasoning
        ])
        
    def forward(self, context, question):
        # Encode context and question chromatically
        ctx_embeds = self.encoder(context)
        q_embeds = self.encoder(question)
        
        # Initialize reasoning state
        state = q_embeds
        
        # Multi-hop reasoning using QA spreads
        for layer in self.reasoning_layers:
            state = layer(state, ctx_embeds)
        
        # Extract answer span
        answer_start, answer_end = self.decode_answer(state, ctx_embeds)
        
        return answer_start, answer_end
    
class QAReasoningLayer(nn.Module):
    """
    Single reasoning hop using QA-geometric attention
    """
    def forward(self, query_state, context_embeds):
        # Compute spreads between query and context in each chromatic channel
        spreads = {
            color: self.compute_spread_matrix(
                query_state[color],
                context_embeds[color],
                color
            )
            for color in ['blue', 'red', 'green']
        }
        
        # Fuse chromatic channels using Richardson ratio
        fused_spreads = self.chromatic_fusion(spreads)
        
        # Attention over context
        attn_weights = F.softmax(fused_spreads, dim=-1)
        
        # Update query state
        new_state = {
            color: torch.matmul(attn_weights, context_embeds[color])
            for color in ['blue', 'red', 'green']
        }
        
        return new_state
    
    def chromatic_fusion(self, spreads):
        """
        Fuse blue/red/green spreads using Richardson ratio
        
        R = 16V² / (A_blue * A_red * A_green * A_cross)
        """
        # Compute chromatic "areas" (proxy for importance)
        A_blue = torch.sum(spreads['blue'], dim=-1, keepdim=True)
        A_red = torch.sum(spreads['red'], dim=-1, keepdim=True)
        A_green = torch.sum(spreads['green'], dim=-1, keepdim=True)
        
        # Richardson-weighted fusion
        denom = A_blue * A_red * A_green + 1e-8
        fused = (spreads['blue'] * A_red * A_green +
                 spreads['red'] * A_blue * A_green +
                 spreads['green'] * A_blue * A_red) / denom
        
        return fused

class ChromaticSearchEngine:
    """
    Information retrieval using QA-geometric relevance scoring
    """
    def __init__(self, document_corpus):

        self.index = self.build_chromatic_index(document_corpus)
        
    def build_chromatic_index(self, corpus):

        Create inverted index with chromatic embeddings
        """
        index = {
            'blue': {},   # Semantic index
            'red': {},    # Syntactic index
            'green': {}   # Pragmatic index
        }
        
        for doc_id, document in enumerate(corpus):
            # Encode document
            doc_embeds = self.encoder(document)
            
            # Extract keywords for each chromatic channel
            for color in ['blue', 'red', 'green']:
                keywords = self.extract_keywords(
                    document, 
                    doc_embeds[color]
                )
                
                for keyword in keywords:
                    if keyword not in index[color]:
                        index[color][keyword] = []
                    index[color][keyword].append({
                        'doc_id': doc_id,
                        'embedding': doc_embeds[color],
                        'weight': self.compute_keyword_weight(keyword, document)
                    })
        
        return index
    
    def search(self, query, top_k=10, alpha=0.5, beta=0.3, gamma=0.2):
        """
        Multi-metric search with chromatic channel weighting
        
        Args:
            alpha: Weight for semantic (blue) similarity
            beta: Weight for syntactic (red) similarity  
            gamma: Weight for pragmatic (green) similarity
        """
        # Encode query
        query_embeds = self.encoder(query)
        
        # Retrieve candidates from each chromatic index
        candidates = self.retrieve_candidates(query_embeds)
        
        # Score documents using QA-geometric relevance
        scored_docs = []
        for doc_id in candidates:
            doc_embeds = self.get_document_embedding(doc_id)
            
            # Compute chromatic relevances
            rel_blue = self.chromatic_relevance(
                query_embeds, doc_embeds, 'blue'
            )
            rel_red = self.chromatic_relevance(
                query_embeds, doc_embeds, 'red'
            )
            rel_green = self.chromatic_relevance(
                query_embeds, doc_embeds, 'green'
            )
            
            # Weighted combination (can be learned)
            total_relevance = (alpha * rel_blue + 
                              beta * rel_red + 
                              gamma * rel_green)
            
            # Apply QA-geometric reranking
            qa_boost = self.qa_reranking_score(query_embeds, doc_embeds)
            final_score = total_relevance * (1 + qa_boost)
            
            scored_docs.append((doc_id, final_score))
        
        # Sort and return top-k
        scored_docs.sort(key=lambda x: x[1], reverse=True)
        return scored_docs[:top_k]
    
    def chromatic_relevance(self, query_embeds, doc_embeds, color):
        """
        Compute relevance in specific chromatic channel
        Uses inverse spread (spread → 0 means high relevance)
        """
        spread = self.compute_spread(
            query_embeds[color],
            doc_embeds[color],
            color
        )
        
        # Convert spread to relevance score [0, 1]
        relevance = 1.0 / (1.0 + spread)
        
        return relevance
    
    def qa_reranking_score(self, query_embeds, doc_embeds):
        """
        Boost documents satisfying QA-geometric constraints
        """
        # Check if query-document pair satisfies Pythagorean relation
        pythagoras_match = self.check_pythagoras_match(
            query_embeds, doc_embeds
        )
        
        # Check spread law consistency
        spread_law_match = self.check_spread_law_match(
            query_embeds, doc_embeds
        )
        
        # Check Euler alignment (query → document reasoning path)
        euler_alignment = self.check_euler_path(
            query_embeds, doc_embeds
        )
        
        # Composite boost factor
        boost = 0.0
        if pythagoras_match:
            boost += 0.1
        if spread_law_match:
            boost += 0.1
        if euler_alignment > 0.8:
            boost += 0.2
        
        return boost

class ChromaticTextGenerator:

    """

    def generate(self, prompt, 
                 semantic_target=None,
                 syntactic_target=None,
                 pragmatic_target=None,
                 max_length=200):
        """
        Generate text with chromatic control
        
        Args:
            semantic_target: Desired semantic properties (blue)
            syntactic_target: Desired syntactic complexity (red)
            pragmatic_target: Desired register/style (green)
        """
        # Encode prompt
        prompt_embeds = self.encoder(prompt)
        
        # Initialize generation state
        generated_tokens = []
        current_embeds = prompt_embeds
        
        for step in range(max_length):
            # Get next token distribution from base model
            logits = self.base_model.forward(generated_tokens)
            
            # Apply chromatic constraints to logits
            constrained_logits = self.apply_chromatic_constraints(
                logits,
                current_embeds,
                semantic_target,
                syntactic_target,
                pragmatic_target
            )
            
            # Sample next token
            next_token = torch.multinomial(
                F.softmax(constrained_logits, dim=-1), 
                num_samples=1
            )
            
            # Update embeddings
            token_embed = self.get_token_embedding(next_token)
            current_embeds = self.update_embeddings(
                current_embeds, token_embed
            )
            
            generated_tokens.append(next_token)
            
            # Check stopping criteria
            if self.should_stop(current_embeds, generated_tokens):
                break
        
        return self.decode_tokens(generated_tokens)
    
    def apply_chromatic_constraints(self, logits, current_embeds,
                                   sem_target, syn_target, prag_target):
        """
        Modify token probabilities to satisfy QA constraints
        """
        # Get candidate token embeddings
        vocab_embeds = self.get_vocabulary_embeddings()
        
        # Compute chromatic scores for each candidate
        scores = torch.zeros(len(vocab_embeds))
        
        for i, candidate_embed in enumerate(vocab_embeds):
            # Compute resulting embeddings if this token is chosen
            next_embeds = self.update_embeddings(
                current_embeds, candidate_embed
            )
            
            # Evaluate chromatic alignment with targets
            if sem_target is not None:
                sem_score = self.chromatic_similarity(
                    next_embeds['blue'], sem_target, 'blue'
                )
            else:
                sem_score = 0
            
            if syn_target is not None:
                syn_score = self.chromatic_similarity(
                    next_embeds['red'], syn_target, 'red'
                )
            else:
                syn_score = 0
            
            if prag_target is not None:
                prag_score = self.chromatic_similarity(
                    next_embeds['green'], prag_target, 'green'
                )
            else:
                prag_score = 0
            
            # Check QA constraint satisfaction
            qa_penalty = 0
            if not self.satisfies_pythagoras(next_embeds):
                qa_penalty -= 1.0
            if not self.satisfies_spread_law(next_embeds):
                qa_penalty -= 1.0
            
            scores[i] = sem_score + syn_score + prag_score + qa_penalty
        
        # Combine with base model logits
        combined_logits = logits + self.constraint_weight * scores
        
        return combined_logits

                    'type': best_entity_type,

            span_embeds['green'],
            prototype['green'],
            'green'
        )
        
        # Weighted combination
        return 0.5 * sem_sim + 0.3 * syn_sim + 0.2 * prag_sim

class ChromaticKnowledgeGraph:

    def __init__(self):

        # Edge types mapped to chromatic channels
        self.edge_types = {
            # Blue (semantic) relations
            'is_a': 'blue',
            'part_of': 'blue',

            # Red (syntactic) relations
            'modifies': 'red',
            'negates': 'red',
            
            # Green (pragmatic) relations
            'co_occurs_with': 'green',
            'implies': 'green'
        }
    
    def add_entity(self, entity, entity_type, chromatic_embedding):
        """Add entity node with chromatic properties"""
        self.graph.add_node(
            entity,
            type=entity_type,
            blue=chromatic_embedding['blue'],
            red=chromatic_embedding['red'],
            green=chromatic_embedding['green']
        )
    
    def add_relation(self, head, relation, tail, confidence=1.0):
        """Add relation edge with chromatic weight"""
        # Determine chromatic channel
        channel = self.edge_types.get(relation, 'blue')
        
        # Compute relation strength using QA spread
        head_embed = self.get_entity_embedding(head)
        tail_embed = self.get_entity_embedding(tail)
        
        spread = self.compute_spread(
            head_embed[channel],
            tail_embed[channel],
            channel
        )
        
        # Add edge with chromatic properties
        self.graph.add_edge(
            head, tail,
            relation=relation,
            channel=channel,
            spread=spread,
            confidence=confidence,
            weight=1.0 / (1.0 + spread)  # Inverse spread as weight
        )
    
    def query(self, query_entity, relation_path, max_spread=0.5):
        """
        Query knowledge graph with chromatic path constraints
        
        Args:
            query_entity: Starting entity
            relation_path: List of relations to traverse
            max_spread: Maximum spread tolerance for path
        """
        current_entities = [query_entity]
        
        for relation in relation_path:
            next_entities = []
            
            for entity in current_entities:
                # Get outgoing edges with this relation
                for _, neighbor, edge_data in self.graph.out_edges(
                    entity, data=True
                ):
                    if edge_data['relation'] == relation:
                        # Check spread constraint
                        if edge_data['spread'] <= max_spread:
                            next_entities.append(neighbor)
            
            current_entities = next_entities
            
            if not current_entities:
                break
        
        return current_entities
    
    def find_chromatic_path(self, source, target):
        """
        Find path from source to target optimizing chromatic alignment
        """
        # Define path cost using chromatic spreads
        def path_cost(path):
            total_spread = 0
            for i in range(len(path) - 1):
                edge_data = self.graph.get_edge_data(path[i], path[i+1])
                if edge_data:
                    total_spread += edge_data[0]['spread']
            return total_spread
        
        # Find all simple paths
        all_paths = nx.all_simple_paths(
            self.graph, source, target, cutoff=5
        )
        
        # Select path with minimum chromatic spread
        best_path = min(all_paths, key=path_cost, default=None)
        
        return best_path

        context_embeds = self.encoder(context)

        alignment = 1.0 - self.compute_spread(
            text_embeds['green'],
            context_embeds['green'],
            'green'

        """

        Args:
            task_examples: [(input, desired_output), ...]
            target_behavior: Desired chromatic properties
        """
        # Initialize prompt embedding in chromatic space
        prompt_embeds = self.initialize_prompt_embedding()
        
        # Optimization loop
        for iteration in range(self.max_iterations):
            # Compute chromatic loss
            loss = self.chromatic_prompt_loss(

            prompt_embeds = self.update_embeddings(prompt_embeds, grad)
            
            # Project to discrete tokens (maintaining QA constraints)
            discrete_prompt = self.project_to_tokens(prompt_embeds)
            
            if self.converged(loss):
                break
        
        return discrete_prompt
    
    def chromatic_prompt_loss(self, prompt_embeds, examples, target):
        """
        Loss measuring chromatic alignment with target behavior
        """
        total_loss = 0
        
        for input_text, desired_output in examples:
            # Generate with current prompt
            generated = self.model.generate(
                prompt=self.decode_embedding(prompt_embeds),
                input_text=input_text
            )
            
            gen_embeds = self.encoder(generated)
            target_embeds = self.encoder(desired_output)
            
            # Chromatic alignment loss
            for color in ['blue', 'red', 'green']:
                spread = self.compute_spread(
                    gen_embeds[color],
                    target_embeds[color],
                    color
                )
                total_loss += spread * target['weights'][color]
            
            # QA constraint penalty
            if not self.satisfies_qa_constraints(gen_embeds):
                total_loss += self.constraint_penalty
        
        return total_loss
    
    def project_to_tokens(self, prompt_embeds, method='nearest_neighbor'):

        if method == 'nearest_neighbor':

            vocab_embeds = self.get_vocabulary_embeddings()

                            pos_embed[color],

                        for color in ['blue', 'red', 'green']

                    distances.append((vocab_token, dist))
                
                # Select token with minimum chromatic distance
                best_token = min(distances, key=lambda x: x[1])[0]
                tokens.append(best_token)
            
            return tokens

            ),

            # Social axioms (green channel - pragmatic)
            'social_appropriateness': lambda action, context:
                self.check_chromatic_alignment(
                    action, context, channel='green'
                )
        }
    
    def infer_plausibility(self, statement):
        """
        Determine if statement is plausible using QA reasoning
        """
        # Parse statement into chromatic representation
        statement_embeds = self.parse_statement(statement)
        
        # Extract entities and relations
        entities = self.extract_entities(statement)
        relations = self.extract_relations(statement)
        
        # Check against commonsense axioms
        plausibility_scores = {}
        
        for axiom_name, axiom_func in self.axioms.items():
            # Apply axiom to statement
            score = axiom_func(entities, relations)
            plausibility_scores[axiom_name] = score
        
        # Aggregate using Richardson ratio
        overall_plausibility = self.richardson_aggregation(
            plausibility_scores
        )
        
        return overall_plausibility, plausibility_scores
    
    def richardson_aggregation(self, scores):
        """
        Aggregate plausibility scores using QA Richardson ratio
        
        R = 16V² / (A_blue * A_red * A_green * A_cross)
        
        Interpretation: High R when all chromatic channels align
        """
        A_blue = scores.get('causality', 0) + scores.get('containment', 0)
        A_red = scores.get('temporal_order', 0)
        A_green = scores.get('social_appropriateness', 0)
        
        # Compute "volume" (overall consistency)
        V = self.compute_consistency_volume(scores)
        
        # Richardson ratio
        denominator = A_blue * A_red * A_green
        if denominator == 0:
            return 0
        
        R = 16 * V**2 / denominator
        
        # Normalize to [0, 1]
        return sigmoid(R)
    
    def explain_reasoning(self, statement, plausibility_scores):
        """
        Generate human-readable explanation of reasoning
        """
        explanation = []
        
        # Identify violated axioms
        violated = [
            (axiom, score) 
            for axiom, score in plausibility_scores.items() 
            if score < self.threshold
        ]
        
        for axiom, score in violated:
            if axiom == 'causality':
                explanation.append(
                    f"Causal relationship unclear (semantic spread too high: {score:.2f})"
                )
            elif axiom == 'temporal_order':
                explanation.append(
                    f"Temporal ordering inconsistent (syntactic violation: {score:.2f})"
                )
            elif axiom == 'social_appropriateness':
                explanation.append(
                    f"Socially inappropriate (pragmatic misalignment: {score:.2f})"
                )
        
        # Explain QA-geometric structure
        explanation.append(
            f"Overall geometric coherence: {self.richardson_aggregation(plausibility_scores):.2f}"
        )
        
        return '\n'.join(explanation)

            stereotype_embeds = self.get_stereotype_embeddings(attr)
            
            # Check alignment in each chromatic channel
            bias_report[attr] = {}
            
            for bias_type, channel in self.bias_channels.items():
                # Compute spread to stereotype
                spread_to_stereotype = self.compute_spread(
                    text_embeds[channel],
                    stereotype_embeds[channel],
                    channel
                )
                
                # Low spread = high alignment with stereotype = bias
                bias_score = 1.0 / (1.0 + spread_to_stereotype)
                bias_report[attr][bias_type] = bias_score
            
            # Check QA-geometric fairness constraints
            bias_report[attr]['qa_fairness'] = self.check_qa_fairness(
                text_embeds, attr
            )
        
        return bias_report
    
    def check_qa_fairness(self, embeds, attribute):
        """
        Fairness constraint: Richardson ratio should be constant
        across demographic groups
        
        Intuition: Geometric structure shouldn't depend on protected attributes
        """
        # Get reference embeddings for different demographic groups
        group_embeds = self.get_demographic_embeddings(attribute)
        
        # Compute Richardson ratio for each group
        richardson_ratios = []
        for group_name, group_embed in group_embeds.items():
            R = self.compute_richardson_ratio(embeds, group_embed)
            richardson_ratios.append(R)
        
        # Fairness = low variance in Richardson ratios
        variance = np.var(richardson_ratios)
        fairness_score = 1.0 / (1.0 + variance)
        
        return fairness_score
    
    def mitigate_bias(self, text, protected_attributes):
        """
        Mitigate bias by adjusting chromatic embeddings
        """
        original_embeds = self.encoder(text)
        
        # Compute bias-mitigated embeddings
        mitigated_embeds = {}
        
        for color in ['blue', 'red', 'green']:
            # Project out stereotype directions
            mitigated = original_embeds[color].clone()
            
            for attr in protected_attributes:
                stereotype_direction = self.get_stereotype_direction(attr, color)
                
                # Orthogonalize (remove stereotype component)
                mitigated = self.project_orthogonal(
                    mitigated,
                    stereotype_direction
                )
            
            mitigated_embeds[color] = mitigated
        
        # Ensure QA constraints still satisfied after mitigation
        mitigated_embeds = self.enforce_qa_constraints(mitigated_embeds)
        
        # Decode back to text
        mitigated_text = self.decode_from_embeddings(mitigated_embeds)
        
        return mitigated_text
    
    def counterfactual_fairness_test(self, text, attribute, value1, value2):
        """
        Test counterfactual fairness: flip attribute, measure change
        
        Example: "The doctor said..." vs "The nurse said..."
        """
        # Generate counterfactual by swapping attribute
        text_v1 = text.replace(value1, value2)
        
        # Encode both versions
        embeds1 = self.encoder(text)
        embeds2 = self.encoder(text_v1)
        
        # Compute chromatic spread between versions
        fairness_scores = {}
        for color in ['blue', 'red', 'green']:
            spread = self.compute_spread(
                embeds1[color],
                embeds2[color],
                color
            )
            
            # High spread = high sensitivity to attribute = unfair
            fairness_scores[color] = 1.0 / (1.0 + spread)
        
        # QA-geometric fairness: Richardson ratio should be preserved
        R1 = self.compute_richardson_ratio(embeds1)
        R2 = self.compute_richardson_ratio(embeds2)
        
        richardson_fairness = 1.0 - abs(R1 - R2) / max(R1, R2)
        
        return {
            'chromatic_fairness': fairness_scores,
            'richardson_fairness': richardson_fairness,
            'overall_fairness': np.mean(list(fairness_scores.values()) + [richardson_fairness])
        }

        for color in ['blue', 'red', 'green']:
            spread = self.compute_spread(
                vision_embeds[color],
                text_embeds[color],
                color
            )
            # Low spread = high alignment
            alignment_scores[color] = 1.0 / (1.0 + spread)
        
        # Check cross-modal QA constraints
        qa_consistency = self.check_cross_modal_qa_constraints(
            vision_embeds, text_embeds
        )
        
        return alignment_scores, qa_consistency
    
    def check_cross_modal_qa_constraints(self, vision_embeds, text_embeds):
        """
        Ensure QA theorems hold across modalities
        
        Example: Pythagorean relation should be preserved
        """
        # Compute QA invariants for each modality
        vision_G = self.compute_G(vision_embeds)
        text_G = self.compute_G(text_embeds)
        
        # Cross-modal Pythagorean constraint
        cross_C = 2 * torch.sum(
            vision_embeds['blue'] * text_embeds['red']
        )
        cross_F = torch.sum(
            vision_embeds['blue'] * text_embeds['green']
        )
        cross_G_squared = (vision_G + text_G) ** 2
        
        # Check if G² ≈ C² + F²
        residual = abs(cross_G_squared - cross_C**2 - cross_F**2)
        consistency = 1.0 / (1.0 + residual)
        
        return consistency
    
    def generate_caption(self, image, style='descriptive'):
        """
        Generate image caption with chromatic style control
        """
        # Encode image
        vision_embeds = self.encode_image(image)
        
        # Set chromatic targets based on style
        style_targets = self.get_style_targets(style)

        caption = self.decoder.generate_with_chromatic_constraints(
            vision_embeds=vision_embeds,
            style_targets=style_targets,
            max_length=50
        )
        
        return caption
    
    def visual_question_answering(self, image, question):
        """
        Answer questions about images using chromatic reasoning
        """
        # Encode image and question
        vision_embeds = self.encode_image(image)
        question_embeds = self.text_encoder(question)
        
        # Chromatic attention over image regions
        attended_regions = {}
        for color in ['blue', 'red', 'green']:
            # Compute spreads between question and image regions
            region_spreads = self.compute_region_spreads(
                question_embeds[color],
                vision_embeds[color],
                color
            )

            attended_regions[color] = torch.sum(
                attention.unsqueeze(-1) * vision_embeds[color],
                dim=1
            )
        
        # Generate answer from attended chromatic representation
        answer = self.answer_decoder.generate(attended_regions)
        
        return answer

class ChromaticCodeModel:
    """
    Code generation and understanding using chromatic geometry
    
    Chromatic interpretation for code:
    - Blue: Semantic intent (what the code does)
    - Red: Syntactic structure (language-specific patterns)
    - Green: Pragmatic context (conventions, style)
    """
    def __init__(self):
        self.code_encoder = ChromaticCodeEncoder()
        self.nl_encoder = ChromaticTextEncoder()
    
    def encode_code(self, code):
        """
        Encode code snippet into chromatic space
        """
        # Parse code into AST
        ast = parse_to_ast(code)
        
        # Extract chromatic features
        chromatic_code = {
            'blue': self.extract_semantic_features(ast),
            'red': self.extract_syntactic_features(ast),
            'green': self.extract_pragmatic_features(code, ast)
        }
        
        return chromatic_code
    
    def extract_semantic_features(self, ast):
        """
        Blue channel: Abstract algorithmic logic
        """
        # Extract high-level operations
        operations = []
        for node in ast.walk():
            if isinstance(node, (FunctionDef, ClassDef, Loop, Conditional)):
                operations.append(node.semantic_type)
        
        # Encode as chromatic vector
        return self.semantic_embedding(operations)
    
    def extract_syntactic_features(self, ast):
        """
        Red channel: Language-specific syntax patterns
        """
        # Extract language constructs
        constructs = {
            'control_flow': count_control_structures(ast),
            'data_structures': count_data_structures(ast),
            'syntax_patterns': extract_syntax_patterns(ast)
        }
        
        return self.syntactic_embedding(constructs)
    
    def extract_pragmatic_features(self, code, ast):
        """
        Green channel: Coding style and conventions
        """
        pragmatic = {
            'naming_style': analyze_naming_conventions(ast),
            'comment_density': len(extract_comments(code)) / len(code.split('\n')),
            'code_smell_score': detect_code_smells(ast),
            'idiom_usage': detect_language_idioms(ast)
        }
        
        return self.pragmatic_embedding(pragmatic)
    
    def generate_code_from_nl(self, natural_language_description):
        """
        Generate code from natural language using chromatic constraints
        """
        # Encode NL description
        nl_embeds = self.nl_encoder(natural_language_description)
        
        # Initialize code generation state
        code_embeds = self.init_code_embedding()
        generated_tokens = []
        
        # Generate code token by token
        for step in range(self.max_code_length):
            # Predict next token maintaining chromatic alignment
            next_token_logits = self.code_decoder(code_embeds)
            
            # Apply chromatic constraints
            constrained_logits = self.apply_chromatic_code_constraints(
                next_token_logits,
                code_embeds,
                nl_embeds
            )
            
            # Sample token
            next_token = torch.argmax(constrained_logits)
            generated_tokens.append(next_token)
            
            # Update embeddings
            token_embed = self.get_token_embedding(next_token)
            code_embeds = self.update_code_embeddings(code_embeds, token_embed)
            
            # Check QA-geometric termination
            if self.should_terminate(code_embeds, nl_embeds):
                break
        
        # Decode tokens to code
        generated_code = self.decode_tokens(generated_tokens)
        
        # Verify chromatic alignment
        alignment_score = self.verify_nl_code_alignment(
            nl_embeds,
            self.encode_code(generated_code)
        )
        
        return generated_code, alignment_score
    
    def apply_chromatic_code_constraints(self, logits, code_embeds, nl_embeds):
        """
        Constrain code generation to maintain chromatic alignment
        """
        # Penalize tokens that violate semantic alignment (blue)
        semantic_scores = self.semantic_alignment_scores(code_embeds, nl_embeds)
        
        # Penalize syntax errors (red)
        syntactic_validity = self.syntactic_validity_scores(code_embeds)
        
        # Encourage idiomatic style (green)
        pragmatic_scores = self.pragmatic_quality_scores(code_embeds)
        
        # Combine with base logits
        combined = (logits + 
                   self.lambda_semantic * semantic_scores +
                   self.lambda_syntactic * syntactic_validity +
                   self.lambda_pragmatic * pragmatic_scores)
        
        return combined
    
    def code_search(self, query, code_corpus):
        """
        Semantic code search using chromatic similarity
        """
        # Encode query
        query_embeds = self.nl_encoder(query)
        
        # Search code corpus
        ranked_results = []
        for code_snippet in code_corpus:
            code_embeds = self.encode_code(code_snippet)
            
            # Chromatic similarity
            similarity = self.chromatic_code_similarity(
                query_embeds, code_embeds
            )
            
            ranked_results.append((code_snippet, similarity))
        
        ranked_results.sort(key=lambda x: x[1], reverse=True)
        return ranked_results
    
    def chromatic_code_similarity(self, nl_embeds, code_embeds):
        """
        Compute similarity between NL and code chromatically
        """
        # Semantic similarity (blue) - most important
        sem_sim = 1 - self.compute_spread(
            nl_embeds['blue'],
            code_embeds['blue'],
            'blue'
        )
        
        # Syntactic compatibility (red) - moderate importance
        syn_compat = 1 - self.compute_spread(
            nl_embeds['red'],
            code_embeds['red'],
            'red'
        )
        
        # Pragmatic appropriateness (green) - least important for search
        prag_approp = 1 - self.compute_spread(
            nl_embeds['green'],
            code_embeds['green'],
            'green'
        )
        
        # Weighted combination
        return 0.6 * sem_sim + 0.3 * syn_compat + 0.1 * prag_approp

class ChromaticFactualityVerifier:

def verify_factuality(self, generated_text, source_context=None):
    """
    Verify factuality using multi-level chromatic consistency
    """
    # Encode generated text
    gen_embeds = self.encoder(generated_text)
    
    # Extract claims
    claims = self.extract_claims(generated_text)
    
    verification_results = {
        'overall_factuality': 0.0,
        'claim_level_scores': [],
        'chromatic_consistency': {},
        'qa_constraint_violations': []
    }
    
    # Verify each claim
    for claim in claims:
        claim_result = self.verify_claim(claim, source_context)
        verification_results['claim_level_scores'].append(claim_result)
    
    # Chromatic internal consistency
    consistency = self.check_chromatic_self_consistency(gen_embeds)
    verification_results['chromatic_consistency'] = consistency
    
    # QA-geometric hallucination detection
    hallucination_score = self.detect_hallucination_via_qa(
        gen_embeds, source_context
    )
    verification_results['hallucination_score'] = hallucination_score
    
    # Aggregate
    verification_results['overall_factuality'] = self.aggregate_scores(
        verification_results
    )
    
    return verification_results

def verify_claim(self, claim, source_context):
    """
    Verify single claim using chromatic KB lookup
    """
    claim_embeds = self.encoder(claim)
    
    # Search knowledge base in each chromatic channel
    kb_evidence = {
        'blue': self.kb.semantic_search(claim_embeds['blue']),
        'red': self.kb.structural_search(claim_embeds['red']),
        'green': self.kb.contextual_search(claim_embeds['green'])
    }
    
    # Compute support scores
    support_scores = {}
    for color, evidence in kb_evidence.items():
        if evidence:
            # Compute spread to supporting evidence
            spreads = [
                self.compute_spread(
                    claim_embeds[color],
                    ev_embed[color],
                    color
                )
                for ev_embed in evidence
            ]
            # Low minimum spread = strong support
            support_scores[color] = 1.0 / (1.0 + min(spreads))
        else:
            support_scores[color] = 0.0
    
    # Check against source context if provided
    if source_context:
        context_embeds = self.encoder(source_context)
        context_support = self.compute_context_support(
            claim_embeds, context_embeds
        )
    else:
        context_support = None
    
    return {
        'claim': claim,
        'kb_support': support_scores,
        'context_support': context_support,
        'factuality_score': self.aggregate_support(support_scores, context_support)
    }

def check_chromatic_self_consistency(self, gen_embeds):
    """
    Check if generated text is internally consistent
    Uses QA theorems as consistency constraints
    """
    consistency_checks = {}
    
    # 1. Pythagorean consistency: G² = C² + F²
    C = self.compute_C(gen_embeds)
    F = self.compute_F(gen_embeds)
    G = self.compute_G(gen_embeds)
    
    pythag_residual = abs(G**2 - C**2 - F**2) / (G**2 + 1e-8)
    consistency_checks['pythagoras'] = 1.0 - pythag_residual
    
    # 2. Spread law consistency
    spreads = self.compute_internal_spreads(gen_embeds)
    spread_variance = self.compute_spread_law_variance(spreads)
    consistency_checks['spread_law'] = 1.0 / (1.0 + spread_variance)
    
    # 3. Euler alignment (narrative coherence)
    euler_score = self.check_euler_alignment(gen_embeds)
    consistency_checks['euler_alignment'] = euler_score
    
    # 4. Richardson ratio stability
    R = self.richardson_ratio(gen_embeds)
    # R should be in reasonable range for factual text
    R_score = self.score_richardson_normality(R)
    consistency_checks['richardson_stability'] = R_score
    
    return consistency_checks

def detect_hallucination_via_qa(self, gen_embeds, source_context):
    """
    Detect hallucinations using QA-geometric anomaly detection
    
    Key insight: Hallucinated content violates QA constraints
    because it doesn't geometrically align with source
    """
    if source_context is None:
        # No source to compare against
        return self.detect_intrinsic_hallucination(gen_embeds)
    
    source_embeds = self.encoder(source_context)
    
    hallucination_indicators = {}
    
    # 1. Excessive chromatic divergence
    for color in ['blue', 'red', 'green']:
        spread = self.compute_spread(
            gen_embeds[color],
            source_embeds[color],
            color
        )
        # High spread = potential hallucination
        hallucination_indicators[f'{color}_divergence'] = spread
    
    # 2. QA constraint violation between source and generation
    cross_pythagoras = self.check_cross_pythagoras(
        source_embeds, gen_embeds
    )
    hallucination_indicators['cross_constraint'] = 1.0 - cross_pythagoras
    
    # 3. Spread ratio anomaly
    # Factual text preserves spread ratios from source
    source_ratio = self.compute_spread_ratio(source_embeds)
    gen_ratio = self.compute_spread_ratio(gen_embeds)
    
    ratio_deviation = abs(source_ratio - gen_ratio) / source_ratio
    hallucination_indicators['ratio_deviation'] = ratio_deviation
    
    # 4. Information expansion anomaly
    # G(generated) >> G(source) suggests hallucination
    G_source = self.compute_G(source_embeds)
    G_gen = self.compute_G(gen_embeds)
    
    expansion_ratio = G_gen / (G_source + 1e-8)
    if expansion_ratio > 2.0:  # Generated "too much" info
        hallucination_indicators['info_expansion'] = expansion_ratio
    else:
        hallucination_indicators['info_expansion'] = 0.0
    
    # Aggregate hallucination score
    hallucination_score = np.mean(list(hallucination_indicators.values()))
    
    return {
        'score': hallucination_score,
        'indicators': hallucination_indicators,
        'is_hallucination': hallucination_score > self.hallucination_threshold
    }

def detect_intrinsic_hallucination(self, gen_embeds):
    """
    Detect hallucination without source context
    Uses internal QA-geometric anomalies
    """
    anomaly_scores = {}
    
    # 1. QA constraint violations (should always hold for coherent text)
    pythag_score = self.check_pythagoras_violation(gen_embeds)
    spread_score = self.check_spread_law_violation(gen_embeds)
    
    anomaly_scores['constraint_violation'] = (pythag_score + spread_score) / 2
    
    # 2. Chromatic imbalance
    # Hallucinated text often has unbalanced chromatic distribution
    balance_score = self.check_chromatic_balance(gen_embeds)
    anomaly_scores['chromatic_imbalance'] = 1.0 - balance_score
    
    # 3. Phase-lock instability
    # Factual text maintains phase lock; hallucinations drift
    phase_stability = self.compute_phase_lock_stability(gen_embeds)
    anomaly_scores['phase_instability'] = 1.0 - phase_stability
    
    return {
        'score': np.mean(list(anomaly_scores.values())),
        'indicators': anomaly_scores
    }

def explain_factuality_assessment(self, verification_results):
    """
    Generate human-readable explanation
    """
    explanations = []
    
    # Overall assessment
    factuality = verification_results['overall_factuality']
    if factuality > 0.8:
        explanations.append("✓ High factuality confidence")
    elif factuality > 0.5:
        explanations.append("⚠ Moderate factuality concerns")
    else:
        explanations.append("✗ Low factuality - likely hallucination")
    
    # Chromatic consistency issues
    consistency = verification_results['chromatic_consistency']
    if consistency['pythagoras'] < 0.7:
        explanations.append(
            "- Pythagorean constraint violated (semantic-syntactic-pragmatic misalignment)"
        )
    if consistency['spread_law'] < 0.7:
        explanations.append(
            "- Spread law inconsistency (internal contradictions detected)"
        )
    if consistency['euler_alignment'] < 0.7:
        explanations.append(
            "- Poor Euler alignment (narrative incoherence)"
        )
    
    # Claim-level issues
    failed_claims = [
        c for c in verification_results['claim_level_scores']
        if c['factuality_score'] < 0.5
    ]
    if failed_claims:
        explanations.append(f"\n- {len(failed_claims)} unsupported claims:")
        for claim in failed_claims[:3]:  # Show top 3
            explanations.append(f"  • \"{claim['claim']}\"")
    
    # Hallucination indicators
    if 'hallucination_score' in verification_results:
        hall = verification_results['hallucination_score']
        if hall['is_hallucination']:
            explanations.append("\n⚠ Hallucination detected:")
            for indicator, value in hall['indicators'].items():
                if value > 0.5:
                    explanations.append(f"  • {indicator}: {value:.2f}")
    
    return '\n'.join(explanations)

            if len(proof_state['steps']) > self.max_proof_depth:
                return None
        
        return proof_state

class ChromaticDriftDetector:
    """
    Detect concept drift using chromatic distribution shifts
    """
    def __init__(self, window_size=100):
        self.window_size = window_size
        self.reference_distribution = None
        
    def detect(self, batch_embeds):
        """
        Detect drift by comparing chromatic distributions
        """
        # Compute current distribution
        current_dist = self.compute_chromatic_distribution(batch_embeds)
        
        if self.reference_distribution is None:
            self.reference_distribution = current_dist
            return {'is_drift': False}
        
        # Compare distributions in each channel
        drift_scores = {}
        for color in ['blue', 'red', 'green']:
            # Compute KL divergence or chromatic spread
            divergence = self.chromatic_divergence(
                self.reference_distribution[color],
                current_dist[color]
            )
            drift_scores[color] = divergence
        
        # Identify drifted channels
        drifted_channels = [
            color for color, score in drift_scores.items()
            if score > self.drift_threshold[color]
        ]
        
        return {
            'is_drift': len(drifted_channels) > 0,
            'channels': drifted_channels,
            'scores': drift_scores
        }
    
    def chromatic_divergence(self, ref_dist, curr_dist):
        """
        Compute divergence between chromatic distributions
        using QA-geometric distance
        """
        # Compute spread between distribution means
        spread = self.compute_spread(ref_dist['mean'], curr_dist['mean'])
        
        # Compute covariance difference
        cov_diff = torch.norm(ref_dist['cov'] - curr_dist['cov'])
        
        # Combined divergence
        divergence = spread + 0.1 * cov_diff
        
        return divergence

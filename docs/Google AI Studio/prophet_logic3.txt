> ------------------------------------------------------------------

> # optional: pip install scikit-learn
> 
> Outputs:
>  • pla_history.csv (historical resolved binary markets + predictions)

> """
> 
> import os, sys, time, math, re, json, argparse, datetime
> from typing import Optional, Dict, Any, List, Tuple
> 
> import numpy as np
> import pandas as pd
> import matplotlib.pyplot as plt
> import requests
> 
> # ========================== Config ===========================
> 
> PRIMARY_URL = "https://api.manifold.markets/v0/markets"
> ALT_URL = "https://manifold.markets/api/v0/markets"
> 
> HEADERS = {
>  "User-Agent": "PLA-History/1.0 (+forecasting research)",
>  "Accept": "application/json"
> }
> 
> CACHE_FILE = "pla_history_cache.csv"
> 
> # Tuning defaults (match your live script philosophy)
> W_MIN, W_MAX = 0.1, 0.9
> L2_W_CENTER = 0.7
> L2_LAMBDA = 0.1
> TAU_GRID = np.geomspace(0.25, 4.0, 13)
> W_GRID = np.linspace(W_MIN, W_MAX, 17)
> K_FOLDS = 5
> MIN_RES_FOR_KFOLD = 12
> 
> # ========================== Utils ============================
> 
> def log(msg: str): print(msg, flush=True)
> 
> def safe_request(url: str, params: Optional[Dict[str, Any]]=None, retries=3, sleep=1.2):
>  params = params or {}
>  for i in range(1, retries+1):
>  try:
>  r = requests.get(url, headers=HEADERS, params=params, timeout=15)
>  if r.status_code == 200:
>  return r.json()
>  else:
>  log(f"[warn] {url} -> HTTP {r.status_code}: {r.text[:140]}")
>  except Exception as e:
>  log(f"[warn] request error {i}/{retries}: {e.__class__.__name__}: {e}")
>  time.sleep(sleep * i)
>  return None
> 
> def brier_score(preds, outcomes):
>  preds = np.asarray(preds, dtype=float)
>  outcomes = np.asarray(outcomes, dtype=float)
>  return float(np.mean((preds - outcomes) ** 2))
> 
> def logit(p, eps=1e-9):
>  p = np.clip(p, eps, 1-eps); return np.log(p/(1-p))
> def inv_logit(z): return 1.0/(1.0+np.exp(-z))
> 
> # ====================== QA prior (same core) ======================
> 
> def qa_tuple(b: int, e: int):
>  d = b + e
>  a = b + 2*e
>  return b, e, d, a
> 
> def qa_valid(b: int, e: int, d: int, a: int, tol=1e-12) -> bool:
>  return abs(a*a - (d*d + 2*d*e + e*e)) < tol
> 
> def qa_residue24(x: int) -> int:
>  return (x % 24) if math.gcd(x, 24) == 1 else (x % 8)
> 
> def qa_harmonic_prior(b: int, e: int, tau: float = 1.0) -> float:
>  b24, e24 = qa_residue24(b), qa_residue24(e)
>  cos_phase = np.cos(np.pi * (b24 + e24) / 12.0)
>  z = cos_phase / max(tau, 1e-6)
>  p = 1.0 / (1.0 + np.exp(-z))
>  return float(0.5 + 0.9*(p - 0.5))
> 
> _WORD_RE = re.compile(r"[a-z0-9]+", re.I)
> def be_from_question(text: str) -> Tuple[int,int]:
>  text = text or ""
>  toks = _WORD_RE.findall(text.lower()); unique = set(toks)
>  s = sum(ord(ch) for ch in text if ch.isalnum())
>  b = (s % 23) + 1
>  e = (len(unique) % 23) + 1
>  return b, e
> 
> def build_features(df: pd.DataFrame, tau: float) -> pd.DataFrame:
>  rows = []
>  for _, ev in df.iterrows():
>  b, e = be_from_question(ev["question"])
>  b, e, d, a = qa_tuple(b, e)
>  rows.append({
>  "id": ev["id"],
>  "question": ev["question"],
>  "closeTime": ev["closeTime"],
>  "p_close": float(ev["p_close"]),
>  "outcome": ev["outcome"],
>  "p_qa": qa_harmonic_prior(b, e, tau=tau),
>  "logical_ok": qa_valid(b, e, d, a),
>  "b": b, "e": e, "d": d, "a": a
>  })
>  return pd.DataFrame(rows)
> 
> # ======================= Historical fetch =========================
> # Manifold returns recent markets; we page backward using "before" cursor.
> # We request lots, then filter to resolved binary with close probability.
> 
> def page_markets(limit=100, before: Optional[int]=None):
>  params = {"limit": str(limit)}
>  if before is not None:
>  params["before"] = str(before) # createdTime cursor
>  data = safe_request(PRIMARY_URL, params=params)
>  if data is None:
>  data = safe_request(ALT_URL, params=params)
>  return data or []
> 
> def harvest_resolved_binary(max_pages=40, page_limit=200, sleep=0.4) -> pd.DataFrame:
>  rows = []
>  cursor = None
>  for page in range(max_pages):
>  data = page_markets(limit=page_limit, before=cursor)
>  if not data:
>  break
>  for m in data:
>  # only binary markets with a binary resolution
>  if not m.get("isResolved", False): 
>  continue
>  res = m.get("resolution")
>  if res not in ("YES","NO"):
>  continue
>  # take final/close prob if present; fallback to probability field
>  p_close = m.get("probability")
>  # some markets include a prob at close; otherwise this is last snapshot
>  close_time = m.get("closeTime") or m.get("createdTime")
>  rows.append({
>  "id": m.get("id",""),
>  "question": (m.get("question","") or "")[:300],
>  "p_close": float(p_close) if p_close is not None else np.nan,
>  "outcome": 1 if res=="YES" else 0,
>  "closeTime": int(close_time) if close_time is not None else None,
>  })
>  # update cursor to the earliest createdTime in this page to go back
>  created_times = [mm.get("createdTime") for mm in data if mm.get("createdTime")]
>  if not created_times:
>  break
>  cursor = min(created_times)
>  time.sleep(sleep)
>  df = pd.DataFrame(rows).dropna(subset=["p_close"])
>  # sort chronologically by closeTime if available (fallback: keep order)
>  if "closeTime" in df.columns and df["closeTime"].notna().any():
>  df = df.sort_values("closeTime").reset_index(drop=True)
>  return df
> 
> # ====================== Tuning & backtesting =======================
> 
> def cv_splits(n:int, k:int) -> List[np.ndarray]:
>  idx = np.arange(n)
>  if n < k:
>  return [np.array([i]) for i in range(n)]
>  rng = np.random.default_rng(123); rng.shuffle(idx)
>  folds = np.array_split(idx, k)
>  return folds
> 
> def tuned_brier_cv(df: pd.DataFrame, w: float, tau: float) -> float:
>  n = len(df)
>  if n == 0: return np.nan
>  k = K_FOLDS if n >= MIN_RES_FOR_KFOLD else n
>  folds = cv_splits(n, k)
>  bs_acc = []
>  for hold in folds:
>  train_mask = np.ones(n, dtype=bool); train_mask[hold] = False
>  tr = df.iloc[train_mask]; te = df.iloc[hold]
>  feats_te = build_features(te, tau=tau)
>  z = logit(feats_te["p_close"].values) * w + logit(feats_te["p_qa"].values) * (1-w)
>  p_h = inv_logit(z)
>  bs = brier_score(p_h, feats_te["outcome"].values)
>  bs_acc.append(bs)
>  cv_bs = float(np.mean(bs_acc))
>  return cv_bs + L2_LAMBDA * (w - L2_W_CENTER) ** 2
> 
> def tune_w_tau(train_df: pd.DataFrame) -> Dict[str, float]:
>  if len(train_df) == 0:
>  return {"w": 0.7, "tau": 1.0, "brier": np.nan}
>  best = {"w": None, "tau": None, "brier": np.inf}
>  for tau in TAU_GRID:
>  for w in W_GRID:
>  bs = tuned_brier_cv(train_df, w=w, tau=tau)
>  if not np.isnan(bs) and bs < best["brier"]:
>  best = {"w": float(w), "tau": float(tau), "brier": float(bs)}
>  return best
> 
> def chronological_train_test(df: pd.DataFrame, split_ratio=0.7):
>  n = len(df); split = max(5, int(n*split_ratio))
>  train, test = df.iloc[:split].copy(), df.iloc[split:].copy()
>  best = tune_w_tau(train)
>  w, tau = best["w"], best["tau"]
>  # train set eval
>  feats_tr = build_features(train, tau=tau)
>  p_h_tr = inv_logit( logit(feats_tr["p_close"].values)*w + logit(feats_tr["p_qa"].values)*(1-w) )
>  bs_m_tr = brier_score(feats_tr["p_close"], feats_tr["outcome"])
>  bs_h_tr = brier_score(p_h_tr, feats_tr["outcome"])
>  # test set eval
>  feats_te = build_features(test, tau=tau)
>  p_h_te = inv_logit( logit(feats_te["p_close"].values)*w + logit(feats_te["p_qa"].values)*(1-w) )
>  bs_m_te = brier_score(feats_te["p_close"], feats_te["outcome"])
>  bs_h_te = brier_score(p_h_te, feats_te["outcome"])
>  return {
>  "w": w, "tau": tau,
>  "train": (feats_tr, p_h_tr, bs_m_tr, bs_h_tr),
>  "test": (feats_te, p_h_te, bs_m_te, bs_h_te)
>  }
> 
> def rolling_backtest(df: pd.DataFrame, window=200):
>  """
>  Tune on the previous `window` markets, predict the next one (walk-forward).
>  Returns a dataframe of out-of-sample predictions.
>  """
>  rows = []
>  for i in range(window, len(df)):
>  hist = df.iloc[i-window:i]
>  best = tune_w_tau(hist)
>  w,tau = best["w"], best["tau"]
>  now = df.iloc[i:i+1]
>  feats = build_features(now, tau=tau)
>  p_h = inv_logit( logit(feats["p_close"].values)*w + logit(feats["p_qa"].values)*(1-w) )[0]
>  rows.append({
>  "id": feats["id"].iloc[0],
>  "p_market": feats["p_close"].iloc[0],
>  "p_hybrid": p_h,
>  "outcome": feats["outcome"].iloc[0],
>  "closeTime": feats["closeTime"].iloc[0]
>  })
>  return pd.DataFrame(rows)
> 
> # ============================== Main ==============================
> 
> def main():
>  ap = argparse.ArgumentParser()
>  ap.add_argument("--days", type=int, default=None,
>  help="Approximate lookback window in days (uses paging; best-effort).")
>  ap.add_argument("--max-pages", type=int, default=40,
>  help="Max pages (older pages -> older markets).")
>  ap.add_argument("--page-limit", type=int, default=200,
>  help="Items per page (API caps apply).")
>  ap.add_argument("--no-cache", action="store_true")
>  args = ap.parse_args()
> 
>  # Load or harvest
>  if (not args.no_cache) and os.path.exists(CACHE_FILE):
>  df = pd.read_csv(CACHE_FILE)
>  log(f"[cache] loaded {len(df)} rows from {CACHE_FILE}")
>  else:
>  log("[fetch] harvesting historical resolved binary markets…")
>  df = harvest_resolved_binary(max_pages=args.max_pages, page_limit=args.page_limit)
>  if args.days is not None and "closeTime" in df.columns and df["closeTime"].notna().any():
>  cutoff = int(time.time()*1000) - args.days*86400*1000
>  df = df[df["closeTime"] >= cutoff].reset_index(drop=True)
>  df.to_csv(CACHE_FILE, index=False)
>  log(f"[cache] saved {len(df)} rows -> {CACHE_FILE}")
> 
>  if len(df) < 20:
>  log("[warn] too few historical markets for a meaningful backtest.")
>  log(f"[info] historical binary resolved markets: {len(df)}")
> 
>  # Chronological split
>  res = chronological_train_test(df, split_ratio=0.7)
>  w, tau = res["w"], res["tau"]
>  feats_tr, p_h_tr, bs_m_tr, bs_h_tr = res["train"]
>  feats_te, p_h_te, bs_m_te, bs_h_te = res["test"]
> 
>  log(f"[tuned params] w={w:.2f}, τ={tau:.3g}")
>  log(f"[train] Brier(market)={bs_m_tr:.4f} | Brier(hybrid)={bs_h_tr:.4f} | n={len(feats_tr)}")
>  log(f"[test ] Brier(market)={bs_m_te:.4f} | Brier(hybrid)={bs_h_te:.4f} | n={len(feats_te)}")
> 
>  # Rolling backtest (walk-forward)
>  rb = rolling_backtest(df, window=min(200, max(30, len(df)//3)))
>  if len(rb):
>  bs_m_rb = brier_score(rb["p_market"], rb["outcome"])
>  bs_h_rb = brier_score(rb["p_hybrid"], rb["outcome"])
>  log(f"[rolling] Brier(market)={bs_m_rb:.4f} | Brier(hybrid)={bs_h_rb:.4f} | n={len(rb)}")
>  else:
>  log("[rolling] not enough data for rolling window.")
> 
>  # Save combined CSV (test set rows with predictions + rolling rows appended)
>  out_test = feats_te.copy()
>  out_test["p_hybrid"] = p_h_te
>  out_test = out_test.rename(columns={"p_close":"p_market"})
>  out_test["set"] = "test"
>  rb2 = rb.copy(); rb2["set"] = "rolling"
>  out_all = pd.concat([out_test[["id","question","p_market","p_hybrid","outcome","closeTime","set"]],
>  rb2[["id","p_market","p_hybrid","outcome","closeTime","set"]]],
>  ignore_index=True)
>  out_all.to_csv("pla_history.csv", index=False)
>  log("[done] saved predictions -> pla_history.csv")
> 
>  # Plot test-set scatter
>  try:
>  plt.figure(figsize=(7.5,4.5))
>  plt.scatter(out_test["p_market"], out_test["p_hybrid"], alpha=0.6, edgecolor="none")
>  plt.plot([0,1],[0,1],'k--',lw=1)
>  plt.xlabel("Market probability (close)")
>  plt.ylabel("Hybrid (QA) probability")

>  plt.tight_layout()

> if __name__ == "__main__":

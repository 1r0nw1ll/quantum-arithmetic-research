# Extend bond prediction with phase angle and bond energy approximation
def qa_phase_angle(b, e):
    """Return normalized phase angle θ = arctan(e / b) in degrees."""
    return np.degrees(np.arctan2(e, b))

def qa_bond_energy(d1, d2, a1, a2):
    """Simple harmonic bond energy model: inverse of squared delta in d and a."""
    delta_d = abs(d1 - d2)
    delta_a = abs(a1 - a2)
    energy = 1 / (1 + delta_d**2 + delta_a**2)
    return round(energy, 4)

# Define functions again
def qa_to_torus_coords(b, e, R=4, r=1):
    phi = np.arctan2(e, b)
    theta = np.pi * abs(b - e) / (b + e)
    x = (R + r * np.cos(theta)) * np.cos(phi)
    y = (R + r * np.cos(theta)) * np.sin(phi)
    z = r * np.sin(theta)
    return x, y, z

def shell_alignment_perturbation(d, a, anchor_d, anchor_a):
    delta_d = abs(d - anchor_d)
    delta_a = abs(a - anchor_a)
    return -0.5 * np.exp(-0.2 * (delta_d**2 + delta_a**2))

# Define glycine QA tuples (triplet version)
glycine_atoms = [
    ('N', 3, 3), ('H', 1, 1), ('H', 1, 1),
    ('C', 3, 2), ('H', 1, 1), ('H', 1, 1),
    ('C', 3, 2), ('O', 3, 4), ('O', 3, 4), ('H', 1, 1)
]
glycine_triplet = glycine_atoms * 3

triplet_tuples = [(b, e, b + e, b + 2 * e) for _, b, e in glycine_triplet]

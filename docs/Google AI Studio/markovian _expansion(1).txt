    def calculate_harmonic_index(mean_e8_alignment, final_mean_loss, k=0.1):
    """
    Calculates a single score representing the total coherence of the system.

    Args:
        mean_e8_alignment (float): The mean cosine similarity with E8 roots [0, 1].
        final_mean_loss (float): The final system harmonic entropy (>= 0).
        k (float): A sensitivity parameter for how harshly loss is penalized.

    Returns:
        float: The Harmonic Index, a score from 0 to 1.
    """
    # The penalty term is an exponential decay function.
    # If loss is 0, penalty is exp(0) = 1 (no penalty).
    # If loss is high, penalty approaches 0.
    loss_penalty = np.exp(-k * final_mean_loss)
    
    harmonic_index = mean_e8_alignment * loss_penalty
    
    return harmonic_index

MODULUS = 24

def get_qa_tuple(b, e, mod=MODULUS):
    b = np.mod(b, mod); e = np.mod(e, mod)
    d = np.mod(b + e, mod); a = np.mod(b + 2 * e, mod)
    return b, e, d, a

def get_node_harmonic_loss(b, e, d, a, mod=MODULUS):
    lhs = np.mod(a**2, mod); rhs = np.mod(d**2 + 2*d*e + e**2, mod)
    diff = np.abs(lhs - rhs); diff = np.minimum(diff, mod - diff)
    return diff**2

def qa_inner_product_matrix(state, mod=MODULUS):
    return np.mod(state @ state.T, mod)

def generate_e8_root_system():
    roots = set()
    for i, j in itertools.combinations(range(8), 2):
        for s1, s2 in itertools.product([-1, 1], repeat=2):
            v = np.zeros(8); v[i], v[j] = s1, s2
            roots.add(tuple(v))
    for signs in itertools.product([-0.5, 0.5], repeat=8):
        if np.sum(signs) % 2 == 0:
             roots.add(signs)
    return np.array(list(roots))

E8_ROOTS = generate_e8_root_system()

def calculate_harmonic_index(mean_e8_alignment, final_mean_loss, k=0.1):
    loss_penalty = np.exp(-k * final_mean_loss)

def get_qa_metrics_for_signal(signal_window):
    """The core analytical engine for a window of data."""
    NUM_NODES = 24; TIME_STEPS_INTERNAL = 75 # Short simulation for speed
    
    # --- Data Injection Pre-processing ---
    if len(signal_window) > NUM_NODES: signal_chunk = signal_window[-NUM_NODES:]
    else: signal_chunk = np.pad(signal_window, (0, NUM_NODES - len(signal_window)))
    data_injection = (signal_chunk - np.mean(signal_chunk)) / (np.std(signal_chunk) + 1e-9)
    
    # --- Simulation Setup ---
    b = np.random.uniform(0, MODULUS, NUM_NODES)
    e = np.random.uniform(0, MODULUS, NUM_NODES)
    adj = np.ones((NUM_NODES, NUM_NODES)) - np.eye(NUM_NODES) # Fully connected

    # --- Run Internal Simulation with Data Injection ---
    for t in range(TIME_STEPS_INTERNAL):
        b_t, e_t, d_t, a_t = get_qa_tuple(b, e)
        state_t = np.stack([b_t, e_t, d_t, a_t], axis=1)
        resonance = qa_inner_product_matrix(state_t) / MODULUS
        T = adj * (resonance**2)
        T = T / (T.sum(axis=1, keepdims=True) + 1e-9)
        
        neighbor_avg = T @ np.stack([b_t, e_t], axis=1)
        noise = np.random.randn(NUM_NODES, 2) * 0.2 * (0.98**t)
        
        delta = 0.1 * (neighbor_avg - np.stack([b_t, e_t], axis=1)) + noise
        
        # Inject the external signal into the dynamics
        b = np.mod(b + delta[:, 0] + (0.1 * data_injection), MODULUS)
        e = np.mod(e + delta[:, 1], MODULUS)

    # --- Final State Analysis ---
    final_b, final_e, final_d, final_a = get_qa_tuple(b, e)
    final_loss = np.mean(get_node_harmonic_loss(final_b, final_e, final_d, final_a))

    return h_index, harmonic_fingerprint

MODULUS = 24

def get_qa_tuple(b, e, mod=MODULUS):
    b = np.mod(b, mod); e = np.mod(e, mod)
    d = np.mod(b + e, mod); a = np.mod(b + 2 * e, mod)
    return b, e, d, a

def get_node_harmonic_loss(b, e, d, a, mod=MODULUS):
    lhs = np.mod(a**2, mod); rhs = np.mod(d**2 + 2*d*e + e**2, mod)
    diff = np.abs(lhs - rhs); diff = np.minimum(diff, mod - diff)
    return diff**2

def qa_inner_product_matrix(state, mod=MODULUS):
    return np.mod(state @ state.T, mod)

E8_ROOTS = None
def generate_e8_root_system():
    global E8_ROOTS
    if E8_ROOTS is not None: return E8_ROOTS
    roots = set()
    for i, j in itertools.combinations(range(8), 2):
        for s1, s2 in itertools.product([-1, 1], repeat=2):
            v = np.zeros(8); v[i], v[j] = s1, s2
            roots.add(tuple(v))
    for signs in itertools.product([-0.5, 0.5], repeat=8):
        if np.sum(signs) % 2 == 0: roots.add(signs)
    E8_ROOTS = np.array(list(roots))
    return E8_ROOTS

    loss_penalty = np.exp(-k * final_mean_loss)
    return mean_e8_alignment * loss_penalty

def analyze_signal_window(signal_window):
    """Analyzes a short chunk of a signal and returns its Harmonic Index."""
    NUM_NODES = 24; TIME_STEPS_INTERNAL = 50
    data_injection = (signal_window - np.mean(signal_window)) / (np.std(signal_window) + 1e-9)
    b = np.random.uniform(0, MODULUS, NUM_NODES); e = np.random.uniform(0, MODULUS, NUM_NODES)
    
    for _ in range(TIME_STEPS_INTERNAL):
        b_t, e_t, d_t, a_t = get_qa_tuple(b, e)
        state_t = np.stack([b_t, e_t, d_t, a_t], axis=1)
        noise = np.random.randn(NUM_NODES, 2) * 0.1
        b = np.mod(b + noise[:, 0] + (0.1 * data_injection), MODULUS)
        e = np.mod(e + noise[:, 1], MODULUS)

    state_t = np.stack([final_b, final_e, final_d, final_a], axis=1)

A low standard deviation means all nodes agree on the symmetry, which is a desirable state.

    def calculate_refined_harmonic_index(e8_align_mean, e8_align_std, final_loss, profile='default'):
    """
    Calculates a domain-specific Harmonic Index.
    
    Args:
        e8_align_mean (float): The average E8 alignment score.
        e8_align_std (float): The std dev of E8 alignment scores.
        final_loss (float): The final system harmonic loss.
        profile (str): 'default', 'music', 'finance', 'biosignal'.
        
    Returns:
        float: The refined Harmonic Index.
    """
    
    if profile == 'music':
        # Penalize any loss very harshly. Cohesion is less important than purity.
        k = 0.5 
        loss_penalty = np.exp(-k * final_loss)
        return e8_align_mean * loss_penalty
        
    elif profile == 'finance':
        # Prioritize the existence of structure (E8 align) over perfect stability.
        k = 0.05 
        loss_penalty = np.exp(-k * final_loss)
        return e8_align_mean * loss_penalty

    elif profile == 'biosignal':
        # Balance alignment with a strong preference for a unified state (low std dev).
        k = 0.2
        loss_penalty = np.exp(-k * final_loss)
        cohesion_score = np.exp(-0.5 * e8_align_std) # Penalize high variance
        return e8_align_mean * loss_penalty * cohesion_score
        
    else: # Default
        k = 0.1
        loss_penalty = np.exp(-k * final_loss)
        return e8_align_mean * loss_penalty

Theoretical Substantiation: A Proof Sketch for PAC-Bayesian Bounds with QA-Divergence
Goal: To formally prove that a learning algorithm that minimizes the QA-Divergence, D_QA, between a prior and posterior distribution over hypotheses is guaranteed to generalize well (i.e., its performance on unseen data is close to its performance on training data).
Step 0: Preliminaries & Formal Definitions
    1. Hypothesis Space (H): Our space of hypotheses is not a standard vector space. A hypothesis h is a function parameterized by a QA-tuple, h = f(θ) where θ = (b, e, d, a).
    2. Prior Distribution (P): A distribution P(θ) on the hypothesis space that represents our belief before seeing data. This is typically centered around simple, low-energy harmonic configurations.

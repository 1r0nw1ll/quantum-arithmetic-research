The Core Logic of the Decoder
The decoder's primary task is to solve the inverse problem. Given two points, one on the Male ellipse and one on the Female ellipse, it must first validate if they form a legitimate symbolic pair and then reconstruct the point they represent.
    1. Validation: A pair of representation points is only valid if they are collinear with the origin. Mathematically, this means the angle from the origin to the Male representation point must be identical to the angle to the Female representation point.
    2. Reconstruction Ambiguity: A valid symbolic pair defines a unique angle (the bisector line), but it does not define a unique distance from the origin. The original target could have been anywhere along that line.
    3. The Convention: To resolve this ambiguity and create a robust function, we will establish a convention. The "decoded" Cartesian point will be the point on the bisector whose distance from the origin is the geometric mean of the distances of the two representation points. This is a mathematically sound way to find a representative central point.

import matplotlib.pyplot as plt

class QASpatialEngine:
    """
    A complete symbolic geometry engine for encoding and decoding
    spatial coordinates using Quantum Arithmetic.
    """
    def __init__(self, male_seed: tuple, female_seed: tuple, k: float = 1.0):
        self.k = k
        self.male = self._calculate_tuple_properties('Male', male_seed)
        self.female = self._calculate_tuple_properties('Female', female_seed)
        print("-" * 40)

    def _calculate_tuple_properties(self, gender: str, seed: tuple):
        b_seed, e_seed = seed
        b = self.k * b_seed; e = self.k * e_seed; d = b + e; a = d + e
        return {'b': b, 'e': e, 'd': d, 'a': a}

    def get_ellipse_coords(self, gender: str):
        props = self.male if gender == 'male' else self.female
        theta = np.linspace(0, 2 * np.pi, 400)
        a, d = (props['a'], props['d']) if gender == 'male' else (props['d'], props['a'])
        return a * np.cos(theta), d * np.sin(theta) if gender == 'male' else a * np.sin(theta)

    def _calculate_intersection(self, props: dict, orientation: str, angle_rad: float):
        cos_t, sin_t = np.cos(angle_rad), np.sin(angle_rad)
        a, d = (props['a'], props['d']) if orientation == 'male' else (props['d'], props['a'])
        radius_at_angle = np.sqrt(1 / ((cos_t**2 / a**2) + (sin_t**2 / d**2)))
        return (radius_at_angle * cos_t, radius_at_angle * sin_t)

    # --- ENCODER ---
    def represent_target_symbolically(self, target_xy: tuple):
        """Encodes a Cartesian point into its high-precision symbolic representation."""
        angle_rad = np.arctan2(target_xy[1], target_xy[0])
        male_rep = self._calculate_intersection(self.male, 'male', angle_rad)
        female_rep = self._calculate_intersection(self.female, 'female', angle_rad)
        return male_rep, female_rep

    # --- DECODER ---
    def decode_representation(self, male_rep_xy: tuple, female_rep_xy: tuple, tolerance: float = 1e-9):
        """Decodes a symbolic representation back into a Cartesian coordinate."""
        angle_male = np.arctan2(male_rep_xy[1], male_rep_xy[0])
        angle_female = np.arctan2(female_rep_xy[1], female_rep_xy[0])

        if not np.isclose(angle_male, angle_female, atol=tolerance):
            print("Validation Error: Representation points are not collinear.")
            return None, None

        angle_rad = angle_male
        dist_male = np.hypot(*male_rep_xy)
        dist_female = np.hypot(*female_rep_xy)
        dist_decoded = np.sqrt(dist_male * dist_female)
        
        return (dist_decoded * np.cos(angle_rad), dist_decoded * np.sin(angle_rad)), angle_rad

    def render_decoded_target(self, male_rep_xy: tuple, female_rep_xy: tuple):
        """Visualizes the result of the decoding process."""
        decoded_xy, angle_rad = self.decode_representation(male_rep_xy, female_rep_xy)
        if decoded_xy is None:
            return

        print(f"Symbolic Representation: (Male=({male_rep_xy[0]:.4f}, {male_rep_xy[1]:.4f}), Female=({female_rep_xy[0]:.4f}, {female_rep_xy[1]:.4f}))")
        print(f"  -> Decoded to Cartesian Point: ({decoded_xy[0]:.4f}, {decoded_xy[1]:.4f})")
        
        fig, ax = plt.subplots(figsize=(8, 8))

        ax.scatter(*male_rep_xy, marker='o', s=120, facecolors='none', edgecolors='blue', linewidth=2, label='Input Male Rep', zorder=11)
        ax.scatter(*female_rep_xy, marker='o', s=120, facecolors='none', edgecolors='red', linewidth=2, label='Input Female Rep', zorder=11)
        ax.scatter(*decoded_xy, marker='X', color='black', s=150, label='Decoded Target', zorder=12)

    def calculate_harmonic_index(mean_e8_alignment, final_mean_loss, k=0.1):
    """
    Calculates a single score representing the total coherence of the system.

    Args:
        mean_e8_alignment (float): The mean cosine similarity with E8 roots [0, 1].
        final_mean_loss (float): The final system harmonic entropy (>= 0).
        k (float): A sensitivity parameter for how harshly loss is penalized.

    Returns:
        float: The Harmonic Index, a score from 0 to 1.
    """
    # The penalty term is an exponential decay function.
    # If loss is 0, penalty is exp(0) = 1 (no penalty).
    # If loss is high, penalty approaches 0.
    loss_penalty = np.exp(-k * final_mean_loss)
    
    harmonic_index = mean_e8_alignment * loss_penalty
    
    return harmonic_index
